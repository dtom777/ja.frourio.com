```ts title="server/api/tasks/controller.ts"
}))
```

```text title="server/prisma/schema.prisma"
model Task {

  label String
  done  Boolean @default(false)
```

[Prisma#findMany](https://www.prisma.io/docs/reference/tools-and-interfaces/prisma-client/crud#findmany)

```ts title="server/service/tasks.ts"
import { PrismaClient } from '@prisma/client'
import { Task } from '$prisma/client' // path alias $prisma -> ./node_modules/.prisma/*

const prisma = new PrismaClient()

export const getTasks = async (limit?: number) =>
  (await prisma.task.findMany()).slice(0, limit)
```

[TypeORM entities](https://typeorm.io/#/entities)

```ts title="server/entity/Task.ts"
import { Entity, PrimaryGeneratedColumn, Column } from 'typeorm'

@Entity()
export class Task {
  @PrimaryGeneratedColumn()
  id: number

  @Column({ length: 100 })
  label: string

  @Column({ default: false })
  done: boolean
}
```

[TypeORM#find](https://typeorm.io/#/find-options)

```ts title="server/service/tasks.ts"
import { getRepository } from 'typeorm'
import { Task } from '$/entity/Task'

const taskRepository = () => getRepository(Task)

export const getTasks = async (limit?: number) =>
  (await taskRepository().find()).slice(0, limit)
```

```ts title="server/service/tasks.ts"
import fs from 'fs'
import { Task } from '$/types'

type DB = {
  nextId: number
  tasks: Task[]
}

const dbPath = 'database.json'

const readDB = async (): Promise<DB> =>
  JSON.parse(await fs.promises.readFile(dbPath, 'utf8'))

if (!fs.existsSync(dbPath)) {
  fs.writeFileSync(dbPath, JSON.stringify({ nextId: 0, tasks: [] }), 'utf8')
}

export const getTasks = async (limit?: number) => (await readDB()).tasks.slice(0, limit)
```
